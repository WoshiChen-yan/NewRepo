FUNCTION get_rssi(target_macs):

    current_mac = None
    FOR line in raw.splitlines():
        IF line.startswith("Station"):

# 通道2：网络层 fping（延迟与丢包）
FUNCTION get_latency(target_ips):
    raw = docker_exec("fping -c 5 -q {all_ips} 2>&1")
    # fping 输出格式:
    # "10.10.10.2 : xmt/rcv/%loss = 5/5/0%, min/avg/max = 1.23/1.56/2.10"

    FOR line in raw.splitlines():
        ip   = regex(r'(\d+\.\d+\.\d+\.\d+)\s*:', line).group(1)
        loss = regex(r'%loss\s*=\s*\d+/\d+/(\d+)%', line).group(1)
        avg  = regex(r'min/avg/max\s*=\s*[\d.]+/([\d.]+)/', line).group(1)

        mac = find_mac_by_ip(ip)     # 遍历全局 nodes 列表反查
        update_link_quality_cache(mac, latency=float(avg), loss=float(loss))

# 两通道数据汇聚到同一 entry_dict：

### 3.3 `get_link_quality_by_mac` 四级查找策略

```
FUNCTION get_link_quality_by_mac(target_mac):

  level 1 (最快): L1 缓存 O(1)
    IF target_mac IN link_quality_cache:
        entry = link_quality_cache[target_mac]
        entry['distance']      = calc_distance(target_node)
        entry['doppler_shift'] = calc_doppler(target_node)
        RETURN entry

  level 2: L2 历史（倒序扫描取最新）
    FOR record in reversed(link_quality_history):
        IF record['mac'] == target_mac:
            RETURN {record + distance + doppler}

  level 3: 纯物理估算（无测量数据）
    IF target_node 存在:
        RETURN {latency=9999, loss=100, rssi=-100, bitrate=0,
                distance=calc, doppler=calc}

  level 4: 完全未知
    RETURN {latency=9999, loss=100, rssi=-100, bitrate=0, distance=0, doppler=0}
```

### 3.4 物理量计算链（伪代码）

```
FUNCTION calculate_siganal_impact(self, target_node):
    # Step 1: 径向相对速度（仅靠近/远离分量产生多普勒效应）
    delta_pos = target_node.position - self.position
    direction_unit = normalize(delta_pos)
    delta_vel = self.direction - target_node.direction
    v_radial = dot(delta_vel, direction_unit)   # 正值=靠近，负值=远离

    # Step 2: 多普勒频移
    f0 = 2.4e9          # 载波频率 2.4 GHz
    c  = 3e8            # 光速
    doppler_hz = (f0 / c) * v_radial   # Δf = (f0/c) * v

    # Step 3: 评估影响
    freq_offset_ratio = abs(doppler_hz) / f0
    IF freq_offset_ratio > 0.01:      # >1% 载波频率
        attenuation = 1 - freq_offset_ratio * 100
    ELSE:
        attenuation = 1.0             # 影响可忽略

    RETURN doppler_hz, attenuation

# 注：当前 attenuation 未被实际用于修改 wmediumd 参数
# 这意味着多普勒效应仅被计算用于特征向量，
# 尚未反馈到物理层仿真（属于未完成的功能）
```

### 4.2 核心节点选举算法（伪代码）

```
FUNCTION select_core_nodes_distributed(num_nodes_rate=0.3):

  # 阶段1：为每个节点计算"局部中心性"得分
  node_scores = {}
  FOR each node in all_nodes:
      score = 0.0

      FOR each target in all_nodes (target != node):
          link = node.get_link_quality_by_mac(target.mac)
          rssi    = link.rssi      # dBm
          loss    = link.loss      # %
          latency = link.latency   # ms

          # 分段线性评分:
          RSSI_THRESHOLD_BASE = -85 dBm
          RSSI_THRESHOLD_GOOD = -65 dBm

          IF rssi <= RSSI_THRESHOLD_BASE:
              link_score = 0          # 信号太差，不贡献分数

          ELIF rssi <= RSSI_THRESHOLD_GOOD:
              link_score = rssi - RSSI_THRESHOLD_BASE    # 0 ~ 20

          ELSE:  # rssi > -65, 质量好，额外奖励
              loss_bonus = (100 - loss) / 10             # 0 ~ 10
              lat_bonus  = 10   if latency < 50ms
                         = 5    if latency < 200ms
                         = 0    otherwise
              link_score = (rssi - RSSI_THRESHOLD_BASE) + loss_bonus + lat_bonus

          score += link_score

      node_scores[node.name] = score

  # 阶段2：Top-K 选举
  num_core = max(1, ceil(N * num_nodes_rate))   # e.g., N=7 -> 2 核心节点
  sorted_nodes = sort(nodes, key=node_scores, descending)

  core_nodes = sorted_nodes[:num_core]
  edge_nodes = sorted_nodes[num_core:]

  # 阶段3：重置健康计数器
  bad_network_health_counter = 0

  # 核心节点角色：
  # core: PPO策略路由 + 参与PPO学习 + 高连通性
  # edge: 执行默认路由（最佳邻居转发）+ 仅存储经验不学习
```

### 4.3 状态向量构建（伪代码）

```
FUNCTION _get_current_state_vector(node):

  # Step 1: 收集所有邻居的链路质量
  all_links = []
  FOR each target in all_nodes (target != node):
      link = node.get_link_quality_by_mac(target.mac)
      all_links.append((link.rssi, target, link))

  # Step 2: 按 RSSI 降序排列，取最强的 MAX_NEIGHBORS 个
  all_links.sort(by rssi, descending)
  top_links = all_links[:MAX_NEIGHBORS]    # MAX_NEIGHBORS = N-1

  # Step 3: 5维特征归一化
  state_vector = []
  neighbor_indices = []
  FOR (rssi, target, link) in top_links:
      state_vector.extend([
          link.distance / 1000.0,    # 距离 m -> km量级
          link.latency  / 100.0,     # 延迟 ms
          link.loss     / 100.0,     # 丢包率 -> [0,1]
          link.rssi     / 100.0,     # 信号强度 dBm (负数)
          link.doppler  / 1000.0,    # 多普勒频移 Hz
      ])
      neighbor_indices.append(index_of(target))

  # Step 4: 零填充（邻居不足时）
  WHILE len(state_vector) < MAX_NEIGHBORS * 5:
      state_vector.append(0.0)

  RETURN state_vector, neighbor_indices
  # neighbor_indices 是 PPO 动作掩码的输入

FUNCTION get_state_sequence(node_name):
  vec, indices = _get_current_state_vector(node)

  IF node_name NOT IN state_history:
      state_history[node_name] = deque(
          [zeros(state_dim)] * seq_len=5,
          maxlen=5
      )

  state_history[node_name].append(vec)     # 滑动窗口自动丢弃最老的
  RETURN np.array(state_history[node_name]), indices
  # 输出形状: [5, state_dim]，供 GRU 处理时序
```

### 4.4 `update_routing` 主训练循环（完整伪代码）

```
FUNCTION update_routing():
  # ================================================================
  # STEP 1: 全节点决策（所有节点都"思考"，不区分核心/边缘）
  # ================================================================
  all_transitions = {}
  all_actions_to_execute = {}

  FOR each node in all_nodes:
      state_seq, nbr_indices = get_state_sequence(node.name)
      actions, logprobs = agent[node.name].select_action(
                              state_seq, node_index, nbr_indices)
      all_transitions[node.name] = (state_seq, actions, logprobs)

      IF node IN core_nodes:
          all_actions_to_execute[node.name] = actions   # 只有核心节点动作被执行

  # ================================================================
  # STEP 2: 路由执行
  # ================================================================
  FOR (node_name, actions) in all_actions_to_execute:
      apply_node_routing(node, actions)   # PPO 策略路由（核心节点）
  FOR edge_node in edge_nodes:
      apply_default_routing(edge_node)    # 贪心路由（边缘节点）

  sleep(0.1)   # 等待内核处理路由更新

  # ================================================================
  # STEP 3: E2E 链路测量（所有节点并发）
  # ================================================================
  test_all_links_concurrent()
  # 为每个节点启动线程：
  #   thread { node.get_latency(all_ips) + node.get_rssi(all_macs) }
  # 测量结果写入所有节点的 link_quality_cache

  # ================================================================
  # STEP 4: 经验收集（所有节点）
  # ================================================================
  global_steps += 1
  core_rewards_this_step = []

  FOR each node in all_nodes:
      (old_state, actions, old_logprobs) = all_transitions[node.name]
      new_state, _ = get_state_sequence(node.name)

      rewards_list = []
      FOR each dest_node in all_nodes:
          IF dest_node == node:
              rewards_list.append(0)
              CONTINUE
          link = node.get_link_quality_by_mac(dest_node.mac)
          r = _calculate_reward_from_link(link)
          rewards_list.append(r)
          reward_history[node.name][dest_node.name].append(r)

      agent[node.name].store(old_state, actions, rewards_list,
                             new_state, old_logprobs, done=False)

      IF node IN core_nodes:
          core_rewards_this_step.append(mean(rewards_list))

  # ================================================================
  # STEP 5: 触发学习（仅核心节点，每 10 步）
  # ================================================================
  IF global_steps > batch_size AND global_steps % 10 == 0:
      FOR core in core_nodes:
          agent[core.name].learn()

  # ================================================================
  # STEP 6: 健康检查与重选举
  # ================================================================
  avg_reward = mean(core_rewards_this_step)
  avg_reward_history.append(avg_reward)

  IF avg_reward < REELECTION_THRESHOLD(-2.0):
      bad_health_counter += 1
  ELSE:
      bad_health_counter = 0

  IF bad_health_counter >= REELECTION_PATIENCE(3):
      # 连续3步平均奖励低于-2.0 -> 触发重选举
      test_all_links_concurrent()       # 先刷新测量数据
      select_core_nodes_distributed()   # 重新选举核心节点（内部会重置counter=0）
```

### 4.5 路由下发的三种情况

```
apply_node_routing(core_node, actions_matrix):
  清除旧路由： ip route flush all proto 188

  FOR each (dest_node, primary_hop_idx, backup_hop_idx) in actions_matrix:

    情况 A: primary_hop == core_node（智能体选了自己）
      -> 不安装路由，忽略
      -> 智能体因得不到奖励信号而学习不选自己

    情况 B: primary_hop == dest_node（直连，1跳可达）
      -> ip route replace <dest_ip> dev wlan<name> metric 10 proto 188
      （直连路由：不需要 via，指定出接口即可）

    情况 C: 标准网关转发（多跳）
      -> 主路：ip route replace <dest_ip> via <primary_ip> metric 10 proto 188
      -> 备路：ip route add    <dest_ip> via <backup_ip>  metric 20 proto 188
      （内核当主路不可达时自动切换到 metric=20 的备用路由）

apply_default_routing(edge_node):
  候选列表按 RSSI 排序

  寻找最佳核心邻居（优先）:
    best_core_nbr = argmax(rssi) among core_nodes within neighbors
    IF found:
        ip route replace default via <best_core_ip> proto 189
        RETURN

  回退到任意最佳邻居:
    best_nbr = argmax(rssi) among all neighbors
    IF found:
        ip route replace default via <best_ip> proto 189
        RETURN

  无邻居（孤立节点）:
    NOT setting default route (孤立节点)

# proto 188 vs proto 189 的分离设计：
# - flush proto 188 只清除核心节点路由，不影响边缘节点默认路由
# - 两类路由互不干扰，安全可控
```

## 5. 子系统 D：强化学习决策层

### 5.1 PPO 网络架构图

```
+======================================================================+
|                  PPOActor（每节点独立，共 N 个）                        |
|                                                                      |
|  输入: state_seq  [batch, seq_len=5, state_dim=N*5]                  |
|                                                                      |
|  +--------------------------------------------+                     |
|  |  GRU Layer                                 |                     |
|  |  input_size = state_dim                    |  hidden_dim=64      |
|  |  捕获时序中的链路质量变化趋势                  |                     |
|  +--------------------+-----------------------+                     |
|          取最后时间步输出 last_hidden [batch, 64]                      |
|                       |                                              |
|  +--------------------v-----------------------+                     |
|  |  Linear(64 -> 128) + ReLU                 |                     |
|  |  Linear(128 -> N * 2 * N)                 |                     |
|  +--------------------+-----------------------+                     |
|                       | reshape                                      |
|  输出: logits [batch, num_dests=N, 2, num_next_hops=N]               |
|         [:, d, 0, :] = 到目标d的"主路径"下一跳概率分布（N选1）          |
|         [:, d, 1, :] = 到目标d的"备路径"下一跳概率分布（N选1）          |
+======================================================================+

+======================================================================+
|              PPOCritic（全局共享，只有 1 个）                           |
|                                                                      |
|  输入A: x_state   [batch, seq_len=5, state_dim]                      |
|  输入B: x_dest_idx [batch]  (Long, 0~N-1, 目的地索引)                 |
|                                                                      |
|  +-------------------------+  +---------------------------+          |
|  |  GRU(state_dim -> 64)   |  |  Embedding(num_dests, 16) |          |
|  +------------+------------+  +------------+--------------+          |
|  last_out [batch, 64]          dest_emb [batch, 16]                  |
|              +------------------------------+                        |
|                         Concat [batch, 80]                           |
|              +------------------------------+                        |
|              |  FC: 80 -> 128  ReLU         |                        |
|              |  FC: 128 -> 64  ReLU         |                        |
|              |  FC: 64 -> 1                 |                        |
|              +---v--------------------------+                        |
|  输出: V(s, dest) [batch, 1]                                          |
|         表示：在当前网络状态s下，到目标dest方向的路由质量价值估计          |
+======================================================================+

设计含义:
  Actor: 每节点独立策略，基于当前节点的局部视图做决策
  Critic: 全局共享，从"整体网络质量"的角度评判价值
  Embedding: 让Critic区分"到不同目标的价值"是不同的
  GRU: 两者都有GRU，让决策基于历史趋势而非瞬时状态
```

### 5.2 动作掩码机制（伪代码）

```
FUNCTION select_action(state_seq, self_index, neighbor_indices):
    logits = actor(state_seq)     # [1, num_dests, 2, N]
    actions = []
    logprobs = []

    FOR dest_idx in range(num_dests):
        FOR path_type in [0=primary, 1=backup]:

            path_logits = logits[0, dest_idx, path_type]  # shape [N]

            # ----- 动作掩码 -----------------------------------------------
            mask = fill(N, value=-inf)         # 初始：全部禁止

            IF neighbor_indices is not empty:
                mask[neighbor_indices] = 0.0   # 解禁：实际邻居

            mask[self_index] = -inf            # 强制禁止：不能选自己
            # 注：如果 neighbor_indices 为空（孤立节点），mask 全为 -inf
            # --------------------------------------------------------------

            masked_logits = path_logits + mask  # 广播加法

            IF all(masked_logits == -inf):      # 孤立节点
                action  = self_index
                logprob = tensor(-1e9)
            ELSE:
                probs = softmax(masked_logits)
                dist  = Categorical(probs)
                action  = dist.sample()
                logprob = dist.log_prob(action)

            actions.append(action)
            logprobs.append(logprob)

    RETURN reshape(actions, [num_dests, 2]), reshape(logprobs, [num_dests, 2])

# 掩码的作用：
# +-------------------------------------------------------+
# | 没有掩码：                                              |
# |   智能体可能选一个实际无法通信的节点                       |
# |   测量时该链路 loss=100%, latency=9999                  |
# |   reward 始终 -10，梯度信号噪声大                        |
# |                                                       |
# | 有掩码：                                                |
# |   只有存在实际无线链路的邻居才能被选中                      |
# |   策略梯度信号更纯净，收敛更快                            |
# +-------------------------------------------------------+
```

### 5.3 PPO 学习过程（完整伪代码）

```
FUNCTION PPOAgent.learn():
    IF len(memory) < batch_size(=5):
        RETURN    # 经验不足，跳过

    batch = random.sample(memory, batch_size)
    # states:       [B, seq_len, state_dim]
    # actions:      [B, num_dests, 2]
    # rewards_list: [B, num_dests]
    # next_states:  [B, seq_len, state_dim]
    # old_logprobs: [B, num_dests, 2]

    logits = actor(states)    # [B, num_dests, 2, N]  一次前向传播

    actor_loss_total  = 0
    critic_loss_total = 0

    FOR dest_idx in range(num_dests):
        dest_idx_batch = [dest_idx] * B   # [B]

        # --- Critic 部分 ---
        V_s  = critic(states,      dest_idx_batch)  # [B]
        V_s_ = critic(next_states, dest_idx_batch)  # [B]
        R    = rewards_list[:, dest_idx]             # [B]

        TD_target = R + gamma(0.99) * V_s_ * (1 - done)
        advantage = (TD_target - V_s).detach()
        critic_loss_total += MSE(V_s, TD_target)

        # --- Actor 部分（主路+备路各一次）---
        FOR path_type in [0, 1]:
            act_this = actions[:, dest_idx, path_type]   # [B]
            old_lp   = old_logprobs[:, dest_idx, path_type]  # [B]

            probs    = softmax(logits[:, dest_idx, path_type])  # [B, N]
            dist     = Categorical(probs)
            new_lp   = dist.log_prob(act_this)    # [B]

            # PPO-Clip 损失
            ratio = exp(new_lp - old_lp)         # 重要性权重
            surr1 = ratio * advantage
            surr2 = clamp(ratio, 1-eps, 1+eps) * advantage   # 截断
            actor_loss_total += -mean(min(surr1, surr2))

    # 归一化
    actor_loss  = actor_loss_total  / (num_dests * 2)
    critic_loss = critic_loss_total / num_dests

    # Actor 独立更新
    optimizerA.zero_grad()
    actor_loss.backward()
    clip_grad_norm_(actor.parameters(), max_norm=0.5)
    optimizerA.step()

    # Critic 共享更新（所有核心节点的learn()都会更新同一个Critic！）
    optimizerC.zero_grad()
    critic_loss.backward()
    clip_grad_norm_(critic.parameters(), max_norm=0.5)
    optimizerC.step()
```

### 5.4 奖励函数设计与分析

```
FUNCTION _calculate_reward_from_link(link_info):

    # 特殊情况 1：完全不可达
    IF link_info IS None:
        RETURN -10.0

    # 特殊情况 2：链路断开（测量超时或全部丢包）
    IF link_info.latency >= 9999.0:
        RETURN -5.0

    latency = min(link_info.latency, 1000.0)
    loss    = min(link_info.loss,    100.0)

    lat_penalty  = (latency / 100.0) * 5.0   # 最大 = 1000/100*5 = 50
    loss_penalty = (loss    / 100.0) * 5.0   # 最大 = 100/100*5 = 5

    RETURN 10.0 - lat_penalty - loss_penalty

奖励值示例：
  最优路径 (latency=0ms,  loss=0%):   reward = +10.0
  典型情况 (latency=10ms, loss=0%):   reward =  +9.5
  普通情况 (latency=20ms, loss=2%):   reward =  +8.9
  较差情况 (latency=200ms,loss=20%):  reward =  -1.0
  极差情况 (latency=500ms,loss=30%):  reward = -16.5   <- 超出[-10,10]范围
  断开情况:                           reward =  -5.0 or -10.0
```

### 6.2 PPO 路由 vs 基准算法的本质区别

```
+----------------------------------------------------------------------+
|                  SPF / ECMP (基准)                                    |
|                                                                      |
|  决策机制：每次从零构建全局图，最优原则基于当前快照                        |
|  时间感知：无（不使用历史数据）                                           |
|  双路备份：ECMP 提供多路等代价路径，但硬件依赖                             |
|  计算代价：O(N^2 log N) Dijkstra                                       |
|  最优性：给定当前链路权重的全局最优解                                      |
|  适用场景：链路质量稳定，预测不重要的场景                                  |
+----------------------------------------------------------------------+

+----------------------------------------------------------------------+
|                  PPO 智能路由                                         |
|                                                                      |
|  决策机制：GRU 处理过去5步历史，感知变化趋势做决策                         |
|  时间感知：有（时序状态，捕获链路质量的变化趋势）                           |
|  双路备份：Primary+Backup 路径，内核层天然支持                           |
|  计算代价：O(actor_inference) per step，无图构建开销                    |
|  最优性：通过采样探索，可能发现反直觉的长期最优策略                         |
|  适用场景：链路动态变化，需要预测性路由的场景                              |
+----------------------------------------------------------------------+
                                                          
关键差异：SPF 是"反应式"，PPO 是"预测式"。
当节点开始移动或信道质量开始退化时，
PPO 可以基于 GRU 的历史编码提前调整路由，
而 SPF 只能在质量已经变差后才能切换。
```

### 7.2 训练数据流

```

  时刻 t（决策前）:
    state_seq_t    [seq_len, state_dim]   <- 历史状态窗口

    logprobs_t     [num_dests, 2]         <- 行为策略对数概率

  时刻 t+1（执行后）:
    state_seq_t1   [seq_len, state_dim]   <- 执行后的新状态
    rewards_t      [num_dests]            <- 每个目的地的奖励

  每 10 步触发 learn():
    从 memory 随机抽取 batch_size=5 条 Transition

      TD_target   = reward_d + 0.99 * V(s',d)

      new_logprob = actor(state_seq) -> dist -> log_prob(action)

## 8. 关键算法深度分析

### 8.1 GRU 在路由决策中的语义意义

```
传统路由（SPF/ECMP）：
    只看"当前时刻链路权重"，决策互相独立

GRU 时序路由：
    state_history = [s_{t-4}, s_{t-3}, s_{t-2}, s_{t-1}, s_t]
                     |过去4步历史                         |当前

    GRU 能隐式学到的时序特征示例：
    · "RSSI最近5步持续下降" -> 预判链路即将断开 -> 提前切换
    · "链路时好时坏，方差大" -> 倾向于选择更稳定的路径
    · "延迟突然升高但RSSI正常" -> 可能是拥塞，预测会恢复
    · "多普勒频移增大" -> 节点在加速运动，信道将退化

    这是 PPO 相比 SPF 的核心优势：时序感知（预测性路由）。
    SPF 是"反应式"（质量变差后才切换），
    PPO 是"预测式"（质量开始退化时提前切换）。
```

### 8.2 核心/边缘双层架构的设计逻辑

```
             节点全集
        +--------------------+
        |    核心节点 30%     |
        |  · 高连通度节点     |      -> PPO 策略路由
        |  · 强信号、低丢包  |      -> 参与学习训练
        |  · 动态可重选举    |      -> 转发骨干
        |         |          |
        +---------|----------+
                  | 核心节点作为边缘节点的网关
        +---------|----------+
        |    边缘节点 70%     |
        |  · 低连通度节点     |      -> 默认路由（最近核心/邻居）
        |  · 弱信号或高丢包  |      -> 仅存储经验，不学习
        |  · 被动跟随        |      -> 通过核心节点转发
        +--------------------+

设计目的:
1. 减少计算量：只训练少数关键节点，边缘节点走简单规则
2. 网络稳定性：边缘节点用贪心路由，不因探索走"奇怪的路"
3. 知识积累：边缘节点也存储经验，若被选为核心节点后可复用
4. 自适应：网络质量下降时自动重选举，令高质量节点担任核心

健康检查触发重选举条件：
    连续 3 步 mean(core_rewards) < -2.0
    -> test_all_links_concurrent(): 刷新数据
    -> select_core_nodes_distributed(): 以最新数据重选
```

### 8.3 共享 Critic 的设计权衡

```
                   对比项            独立Critic     共享Critic（当前方案）
                   --------          ----------     ------------------
                   参数量             N * Critic参   1 * Critic参数量
                   数据利用率         各节点独立      所有节点共同训练
                   全局一致性         无             Critic有统一价值观
                   目的地区分能力      无             Embedding实现
                   并发更新风险        无             存在（N次.step()）
                   边缘节点经验利用    无             部分（future reuse）

共享Critic的隐含假设：
  "不同节点、不同目的地的路由质量价值，可以用同一个网络评估"
  这在网络拓扑对称或节点同质时成立，
  但在节点性能差异较大时可能导致价值估计偏差。
```

### 8.4 双路备份路由与 Linux 内核路由表的交互

```
Linux 路由表结构（ip route show）：
  <dest_prefix> via <gateway> dev <iface> metric <N> proto <P>

说明：
  proto 188 = 本系统核心节点路由（可被 flush proto 188 批量清除）
  proto 189 = 本系统边缘节点默认路由（独立于188，不被flush影响）
  metric    = 路由优先级，数越小越优先
  kernel 内置 failover：主路不通时自动选 metric 更大的备用路

双路路由的生效过程：
  节点A到节点G存在两条路由:
    route 1: 10.10.10.17/32 via 10.10.10.12  metric 10  proto 188  (主路)
    route 2: 10.10.10.17/32 via 10.10.10.14  metric 20  proto 188  (备路)

  正常情况: 内核选 metric=10 的主路 -> 10.10.10.12 -> 继续转发
  主路断开: ARP 超时或 route unreachable -> 内核自动降级到 metric=20
  主路恢复: 下一轮 update_routing() 重新安装两条路由（flush 后重装）

路由安装频率：
  每次 update_routing() 都会 flush + 重装所有 proto 188 路由
  这意味着每步路由都是"从头安装"，不存在"路由沉默"问题
```

## 9. 设计问题与改进方向

### 9.1 已确认的结构性缺陷

```
[BUG 1] memory deque 无上限 -> 内存泄漏
  位置: ppoagent.py PPOAgent.__init__()
  代码: self.memory = deque()           # 缺少 maxlen 参数
  影响: 每步存1条Transition(~KB级别), 长期训练后memory无限增长
  修复: self.memory = deque(maxlen=10000)

------------------------------------------------------------------

[BUG 2] 奖励函数值域超出预期范围
  位置: net.py _calculate_reward_from_link()
  代码: lat_penalty = (latency / 100.0) * 5.0
  影响: latency=500ms 时 lat_penalty=25 -> reward=-16.5 (声称最低为-10)
        Critic 价值估计范围不稳定
  修复: 添加 reward = clip(reward, -10.0, 10.0) 或调整系数

------------------------------------------------------------------

[BUG 3] 共享 Critic 被多次重复更新
  位置: net.py update_routing() Step 5
  代码: for core in core_nodes: agent[core].learn()
        每次 learn() 内部执行 optimizerC.step()
  影响: 3个核心节点 -> Critic 被更新3次，用不同数据batch，可能过拟合
  修复: 将 Critic 更新从 PPOAgent.learn() 中剥离，
        在 Net 层统一聚合批次后执行一次更新

[BUG 4] 边缘节点 memory 无限增长但永不消耗
  位置: net.py update_routing() Step 4

        for core in core_nodes: agent.learn()   # 只有核心学
  影响: 边缘节点的 memory 不断增加但永远不被 sample()
  修复: 边缘节点也使用 maxlen OR 仅对核心节点调用 store()

[BUG 5] 多线程并发写入 link_quality_history，无锁保护
